### Task 3: Create `roles/pi-hole/tasks/main.yml`

**Files:**
- Create: `roles/pi-hole/tasks/main.yml`

- [ ] **Step 3.1: Create `roles/pi-hole/tasks/main.yml`**

```yaml
---
# tasks file for pi-hole

- name: Install dependencies
  ansible.builtin.apt:
    name:
      - curl
      - git
      - ca-certificates
    state: present
    update_cache: true

- name: Check if pi-hole is installed
  ansible.builtin.stat:
    path: /usr/local/bin/pihole
  register: _pihole_installed

- name: Create pi-hole config directory
  ansible.builtin.file:
    path: "{{ pihole_install_dir }}"
    state: directory
    mode: "0755"
  when: not _pihole_installed.stat.exists

- name: Template unattended install config
  ansible.builtin.template:
    src: setupVars.conf.j2
    dest: "{{ pihole_install_dir }}/setupVars.conf"
    mode: "0644"
  when: not _pihole_installed.stat.exists

- name: Download pi-hole installer
  ansible.builtin.get_url:
    url: "https://install.pi-hole.net"
    dest: /tmp/pihole-install.sh
    mode: "0755"
  when: not _pihole_installed.stat.exists

- name: Run pi-hole unattended install
  ansible.builtin.command:
    cmd: bash /tmp/pihole-install.sh --unattended
  environment:
    PIHOLE_SKIP_OS_CHECK: "true"
  when: not _pihole_installed.stat.exists
  changed_when: true
  notify: Restart pihole-FTL

- name: Set pi-hole admin password
  ansible.builtin.command:
    cmd: pihole -a -p "{{ pihole_admin_password }}"
  no_log: true
  changed_when: true
  when: not _pihole_installed.stat.exists

- name: Ensure setupVars.conf is up to date
  ansible.builtin.template:
    src: setupVars.conf.j2
    dest: "{{ pihole_install_dir }}/setupVars.conf"
    mode: "0644"
  notify: Restart pihole-FTL

- name: Ensure pihole-FTL is enabled and running
  ansible.builtin.service:
    name: pihole-FTL
    enabled: true
    state: started
```
